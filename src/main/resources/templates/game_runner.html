Cours Python — Questions du Runner

Fonctions de base

print("Bonjour")      # afficher
nom = input("Nom : ") # saisir
len("Bonjour")        # → 7 (longueur)
int("42")             # → 42 (convertir en entier)
str(42)               # → "42" (convertir en texte)

Listes

ma_liste = [1, 2, 3]
ma_liste.append(4)    # ajouter a la fin → [1,2,3,4]
ma_liste[0]           # → 1 (premier element)
len(ma_liste)         # → 4

Boucles

range(5)              # genere 0,1,2,3,4
for i in range(5):
    print(i)          # s'execute 5 fois

Dictionnaires

dico = {"nom": "Abir", "age": 16}
dico.keys()   # → dict_keys(["nom", "age"])
dico.values() # → dict_values(["Abir", 16])
